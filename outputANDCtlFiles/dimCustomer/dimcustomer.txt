CREATE TABLE dimcustomer(
SK_CustomerID           NUMBER GENERATED by default on null as IDENTITY,
CustomerID              NUMBER NOT NULL,
TaxID                   VARCHAR2(20) NOT NULL,
Status                  VARCHAR2(10) NOT NULL,
LastName                VARCHAR2(30) NOT NULL,
FirstName               VARCHAR2(30) NOT NULL,
MiddleInitial           VARCHAR2(1),
Gender                  VARCHAR2(1),
Tier                    NUMBER,
DOB                     DATE NOT NULL,
AddressLine1            VARCHAR2(80) NOT NULL,
AddressLine2            VARCHAR2(80),
PostalCode              VARCHAR2(12) NOT NULL,
City                    VARCHAR2(25) NOT NULL,
StateProv               VARCHAR2(20) NOT NULL,
Country                 VARCHAR2(24),
Phone1                  VARCHAR2(30),
Phone2                  VARCHAR2(30),
Phone3                  VARCHAR2(30),
Email1                  VARCHAR2(50),
Email2                  VARCHAR2(50),
NationalTaxRateDesc     VARCHAR2(50),
NationalTaxRate         NUMBER,
LocalTaxRateDesc        VARCHAR2(50),
LocalTaxRate            NUMBER,
AgencyID                VARCHAR2(30),
CreditRating            NUMBER,
NetWorth                NUMBER,
MarketingNameplate      VARCHAR2(100),
IsCurrent               VARCHAR2(1 byte) NOT NULL,
BatchID                 NUMBER NOT NULL,
EffectiveDate           DATE NOT NULL,
EndDate                 DATE NOT NULL
);
